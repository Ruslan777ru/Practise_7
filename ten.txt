def fib(n):
    if(n in [1,2]):
        return 1
    return fib(n-1)+fib(n-2)

print(fib(4))

fib_m={
    1:1,
    2:1
}

def fib(n):
    result=fib_m.get(n)
    if(result is None):
        result=fib(n-1)+fib(n-2)
        fib_m[n]=result
    return result

print(fib(200))
